27 namespace Flikore\Validator;

29 /**
30  * Combines two or more validator objects into one. This validates with all
31  * the inserted validators and stops at the first error, return the message
32  * of the validator that went wrong.
33  *
34  * @author George Marques <george at georgemarques.com.br>
35  * @version 0.5.0
36  * @since 0.2
37  * @license http://opensource.org/licenses/MIT MIT
38  * @copyright (c) 2014, George Marques
39  * @package Flikore\Validator
40  */
41 class ValidationCombo extends Validator
42 {
43 
44     /**
45      * The error message for this validator.
46      * @var string The error message for this validator.
47      */
48     protected $message = null;
49 
50     /**
51      * A collection of all validators.
52      * @var Validator[] A collection of all validators.
53      */
54     protected $validators = array();
55 
56     /**
57      * Adds a new validator to the combo.
58      * @param Validator $validator The validator to add.
59      */
60     public function addValidator(Validator $validator)
61     {
62         $validator->addKeyValue('key', $this->values['key']);
63         array_push($this->validators, $validator);
64     }
65 
66     /**
67      * Executes the real validation so it can be reused.
68      * @param mixed $value The value to validate.
69      * @return boolean Whether the value pass the validation.
70      */
71     protected function doValidate($value)
72     {
73         foreach ($this->validators as $rule)
74         {
75             if (!$rule->validate($value))
76             {
77                 $this->setErrorMessage($this->message === null ? $rule->getErrorMessage() : $this->message);
78                 return false;
79             }
80         }
81         return true;
82     }
83 
84     /**
85      * Adds a new key-value pair to be replaced by the templating engine.
86      * This does not check if it's replacing a specific validator value.
87      * @param string $key The key to replace (in the template as "%key%")
88      * @param string $value The value to be inserted instead of the key.
89      */
90     public function addKeyValue($key, $value)
91     {
92         foreach ($this->validators as $v)
93         {
94             $v->addKeyValue($key, $value);
95         }
96     }
97 
98 }
