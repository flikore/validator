 49             }
 50         }

 54          * This should work whether or not there was a test before.

 57         public function getErrorMessage()

 59             return $this->applyTemplate();
 60         }

 71         /**

 75          * @param string $value The value to be inserted instead of the key.

 77         public function addKeyValue($key, $value)
 78         {

 80         }

 82         /**
 83          * Applies the template message to a formed one.
 84          * @return string The formed message.
 85          */
 86         protected function applyTemplate()
 87         {
 88             $message = dgettext('Flikore.Validator', $this->message);
 89             foreach ($this->values as $key => $value)
 90             {
 91                 $message = str_replace("%$key%", dgettext('Flikore.Validator', $value), $message);
 92             }
 93             return $message;
 94         }
