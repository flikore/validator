 27 namespace Flikore\Validator\Validators;
 28 
 29 use Flikore\Validator\Validator;
 30 
 31 /**
 32  * Validates that the value pass in at least one of the given validators.
 33  * 
 34  * @customKey <i>%vN%</i> The message of the Nth validator (1-index).
 35  *
 36  * @author George Marques <george at georgemarques.com.br>
 37  * @version 0.4.0
 38  * @since 0.4.0
 39  * @license http://opensource.org/licenses/MIT MIT
 40  * @copyright (c) 2014, George Marques
 41  * @package Flikore\Validator
 42  */
 43 class OrValidator extends Validator
 44 {
 45 
 46     /**
 47      * An array of validators to test.
 48      * @var Validator[] An array of validators to test.
 49      */
 50     protected $validators = array();
 51 
 52     /**
 53      * The error message for this validator.
 54      * @var string The error message for this validator.
 55      */
 56     protected $message = 'The %key% must match one of the validators.';
 57 
 58     /**
 59      * Creates a new Or Validator. You can pass as many validators as you want.
 60      * 
 61      * @param \Flikore\Validator\Validator $v1 The first validator to check.
 62      * @param \Flikore\Validator\Validator $v2 The second validator to check.
 63      * @param \Flikore\Validator\Validator ...$v The other validators to check.
 64      */
 65     public function __construct(Validator $v1, Validator $v2)
 66     {
 67         $i = 1;
 68         foreach (func_get_args() as $arg)
 69         {
 70             if (!$arg instanceof Validator)
 71             {
 72                 throw new \InvalidArgumentException(dgettext('Flikore.Validator', 'The arguments must be intances of validators'));
 73             }
 74             $this->validators[] = $arg;
 75             $this->addKeyValue('v' . $i++, $arg->getErrorMessage());
 76         }
 77     }
 78 
 79     /**
 80      * Adds a new key-value pair to be replaced by the templating engine.
 81      * This does not check if it's replacing a specific validator value.
 82      * 
 83      * @param string $key The key to replace (in the template as "%key%")
 84      * @param string $value The value to be inserted instead of the key.
 85      */
 86     public function addKeyValue($key, $value)
 87     {
 88         parent::addKeyValue($key, $value);
 89 
 90         if (preg_match('/^v[0-9]+$/', $key) == 0)
 91         {
 92             foreach ($this->validators as $v)
 93             {
 94                 $v->addKeyValue($key, $value);
 95             }
 96         }
 97     }
 98 
 99     /**
100      * Applies the template message to a formed one.
101      * @return string The formed message.
102      */
103     protected function applyTemplate()
104     {
105         $i = 1;
106         foreach ($this->validators as $v)
107         {
108             $this->addKeyValue('v' . $i++, $v->getErrorMessage());
109         }
110         return parent::applyTemplate();
111     }
112 
113     /**
114      * Executes the real validation so it can be reused.
115      * @param mixed $value The value to validate.
116      * @return boolean Whether the value pass the validation.
117      */
118     protected function doValidate($value)
119     {
120         foreach ($this->validators as $v)
121         {
122             if($v->validate($value))
123             {
124                 return true;
125             }
126         }
127         return false;
128     }
129 
130 }
